<?php namespace Modules\User\Entities;

use Modules\Core\Entities\Eloquent\CoreModel;

class ContactLike extends CoreModel {
    /**
     * The table associated with the model.
     * @var string
     */
    protected $table = 'user__contact_likes';

    /**
     * Indicates if the model should be timestamped.
     *
     * @var bool
     */
    public $timestamps = false;

    /**
     * The attributes that are mass assignable.
     * @var array
     */
    protected $fillable = ['user_id', 'contact_id', 'liked'];

    /**
     * The attributes that should be hidden for serialization.
     * @var array
     */
    protected $hidden = [
    ];

    /**
     * The attributes that should be casted to native types.
     * @var array
     */
    protected $casts = [
        'id'         => 'integer',
        'user_id'    => 'integer',
        'contact_id' => 'integer',
        'liked'      => 'boolean',
    ];

    /**
     * Relationship
     */
    public function user() {
        return $this->belongsTo('\Modules\User\Entities\Sentinel\User', 'user_id', 'id');
    }

    public function contact() {
        return $this->belongsTo('Modules\User\Entities\Contact', 'contact_id');
    }
}
