<?php namespace Modules\User\Entities;

use Modules\Core\Entities\Eloquent\CoreModel;

/**
 * Class Follow
 * @package Modules\User\Entities
 * @author Huy D <huydang1920@gmail.com>
 * @copyright (c) Motila Corporation
 */
class Follow extends CoreModel {
    /**
     * The table associated with the model.
     * @var string
     */
    protected $table = 'user__follows';

    /**
     * The attributes that are mass assignable.
     * @var array
     */

    protected $fillable = [
        'user_id',
        'follow_id',
    ];

    protected $hidden = [];

    /**
     * The attributes that should be casted to native types.
     * @var array
     */
    protected $casts = [
        'user_id'   => 'integer',
        'follow_id' => 'integer',
    ];

    public function user() {
        return $this->belongsTo('Modules\User\Entities\Sentinel\User', 'user_id');
    }

    public function follow() {
        return $this->belongsTo('Modules\User\Entities\Sentinel\User', 'follow_id');
    }
}
