<?php

namespace Modules\User\Entities;

use Modules\Core\Entities\Eloquent\CoreModel;

class Collection extends CoreModel {
    /**
     * The table associated with the model.
     *
     * @var string
     */
    protected $table = 'user__collections';

    /**
     * Indicates if the model should be timestamped.
     *
     * @var bool
     */
    public $timestamps = true;

    protected $fillable = [
        'user_id',
        'object',
        'object_id',
        'folder_id'
    ];

    /**
     * The attributes that should be hidden for serialization.
     *
     * @var array
     */
    protected $hidden = [
    ];

    /**
     * The attributes that should be casted to native types.
     *
     * @var array
     */
    protected $casts = [
        'id'        => 'integer',
        'user_id'   => 'integer',
        'object_id' => 'integer',
        'folder_id' => 'integer',
    ];

    /**
     * Relationship
     */
    public function user() {
        return $this->belongsTo('\Modules\User\Entities\Sentinel\User');
    }

    public function folder() {
        return $this->belongsTo('\Modules\User\Entities\CollectionFolder');
    }

    /*public function place() {
        return $this->belongsTo('\Modules\Place\Entities\Place', 'place_id', 'id');
    }*/
}
